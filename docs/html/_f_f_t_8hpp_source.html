147 static std::vector<float> buildHanningWindow(int _windowSize) {
148 std::vector<float> window(_windowSize);
149
150 for(int n = 0; n < _windowSize; ++n) {
151 window[n] = 0.5 * (1.0f - cosf((2.0f * M_PI * n) / (_windowSize -1 )));
152 }
153
154 return window;
155 }

162 static float calculateEnergyWithHanningWindow(
163 std::vector<float> in,
164 int centerBin,
165 std::vector<float> _hanningWindow
166 ) {
167 float energy = 0.0f;
168 int halfWindow = _hanningWindow.size() / 2;
169 for (int n = 0; n < _hanningWindow.size(); ++n) {
170 int bin = centerBin - halfWindow + n;
171 if (bin >= 0 && bin < in.size()) {
172 energy += in[bin] * in[bin] * _hanningWindow[n];
173 }
174 }
175
176 return energy;
177 }
