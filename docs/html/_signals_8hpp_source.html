150 static std::function<float()> getSineWaveCalculator(float freq = 1, float amplitude = 1, float phase=0) {
151 return [freq, amplitude, phase]() {
152 float time = ((float) ofGetElapsedTimef());
153 float result = amplitude * std::sin(2.0 * M_PI * freq * time + phase);
154
155 return result;
156 };
157 }
158
168 static std::function<float()> getSquareWaveCalculator(float freq = 1, float amplitude = 1, float phase=0) {
169 return [freq, amplitude, phase]() {
170 float time = ((float) ofGetElapsedTimef());
171 float sine = amplitude * std::sin(M_PI * freq * time + phase);
172 float result = amplitude * (sine>0 ? 1 : -1);
173
174 return result;
175 };
176 }

184 static std::function<float()> getRandomCalculator(float amplitude=1, float sampleDuration=0) {
185 return [amplitude, sampleDuration]() {
186 static float lastTime;
187 static float result;
188 float diff = ofGetElapsedTimef() - lastTime;
189 if (diff < sampleDuration) {
190 return result;
191 }
192
193 result = ofRandom(amplitude);
194
195 lastTime = ofGetElapsedTimef();
196 return result;
197 };
198 }

207 static std::function<float()> getPulseCalculator(float freq, float amplitude, float dutyCycle) {
208 return [freq, amplitude, dutyCycle]() {
209 float time = ((float) ofGetElapsedTimef());
210 double period = 1.0 / freq;
211 double modTime = fmod(time, period);
212 return (modTime < dutyCycle * period) ? amplitude : 0.0;
213 };
